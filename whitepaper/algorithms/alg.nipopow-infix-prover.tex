\begin{algorithm}[H]
    \caption{\label{alg.nipopow-infix-prover}The \textsf{Prove} algorithm
    for infix proofs}
    \begin{algorithmic}[1]
        \Function{\sf ProveInfix$_{m,k}$}{$\chain$, $\chain'$, \textsf{depth}} % , realLink, blockById}
            \Let{(\pi, \chi)}{\textsf{Prove}_{m,k}(\chain)}
            \For{$B' \in \chain'$}
                \For{$E \in \pi$}
                    \If{$\textsf{depth}[E] \geq \textsf{depth}[B']$}
                        \Let{R}{\textsf{followDown}(E, B', \textsf{depth})} %, \textsf{realLink}, \textsf{blockById})}
                        % \Let{L}{\textsf{followUp}(B', \textit{level}(E'), \textsf{realLink}, \textsf{blockById})}
                        % \Let{aux}{aux \cup L}
                        \Let{aux}{aux \cup R}
                        \State\textbf{break}
                    \EndIf
                    \Let{E'}{E}
                \EndFor
            \EndFor
            \State\Return{$(aux \cup \pi, \chi)$}
        \EndFunction
    \vskip8pt
    \end{algorithmic}
\end{algorithm}
