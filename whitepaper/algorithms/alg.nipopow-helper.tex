\begin{algorithm}[H]
    \caption{\label{alg.nipopow-helper}Helper functions needed for the NIPoPoW
    construction. \textit{id} determines the id of a block given its
    contents. \textit{level} determines the maximum superblock level of a block.}
    \begin{algorithmic}[1]
        \Function{\sf id}{$B$}
            \Let{(s, x, ctr, \textsf{interlink})}{B}
            \Let{\textsf{id}}{H(ctr, G(s, x, \textsf{interlink}))}
            \State\Return{\textsf{id}}
        \EndFunction

        \Function{\it level}{$B$}
            \Let{\mu}{\left \lfloor \log(T) - \log(\sf{blockid}(B)) \right \rfloor}
            \State\Return{$\mu$}
        \EndFunction
    \vskip8pt
    \end{algorithmic}
\end{algorithm}
